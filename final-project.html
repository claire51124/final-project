Final Project¶

I divide my final project into 3 sections.

1. Strategy and Backtest¶

In the section 1, I select largest 1000 firms, then select the smallest 800 firms, and then select the intersection of 600 firms withe highest PE ratio and 600 firms with highest PB ratio.

I use market weight and rebalance quarterly.

2. Sharpe ratio¶

By comparing the mcap weight and equal weight, try to figure which weight could make portfolio has the higher sharpe ratio.

3. Spread¶

By adding th spread into my portfolio, and find that the return of my strategy is lower than the return of SPY.

def get_rebalance_dates(frequency):
    if   frequency == 'A': group = PRICE.groupby( PRICE.index.year)
    elif frequency == 'Q': group = PRICE.groupby([PRICE.index.year, PRICE.index.quarter])  
    elif frequency == 'M': group = PRICE.groupby([PRICE.index.year, PRICE.index.month])    
    elif frequency == 'W': group = PRICE.groupby([PRICE.index.year, PRICE.index.week])     
    return PRICE.index[:1].union(group.tail(1).index) 

def calc_returns(start, end, assets):
    price_ret = PRICE.loc[start:end,assets].ffill().pct_change()
    return RET.loc[start:end,assets][1:].fillna(price_ret).add(1).cumprod() 


def run_backtest(frequency):
    rebalance_dates = get_rebalance_dates(frequency)    
    portfolio_value = pd.Series(1,index=[rebalance_dates[0]])

    for i in range(len(rebalance_dates)-1):
        start_date = rebalance_dates[i]
        end_date   = rebalance_dates[i+1]
        assets_we_can_buy = PRICE.loc[start_date].dropna().index
        assets            = select_assets(start_date, assets_we_can_buy)
        weights           = select_weights(start_date, assets)
        cum_ret               = calc_returns(start_date, end_date, assets)
        asset_values          = portfolio_value[-1] * weights  * cum_ret  
        portfolio_value       = portfolio_value.append(asset_values.sum(axis=1)) 
        
    return portfolio_value

def equal_weight_sharpe_ratio(start, end, assets):
    r_daily         = PRICE.loc[start:end,assets].ffill().pct_change()   #portfolio return
    r_quarterly     = r_daily[end:end]
    shy_daily       = shy[start:end].ffill().pct_change()                #riskfree return
    shy_quarterly   = shy_daily[end:end]
    rx_daily        = r_daily.sub( shy_daily, axis=0 )                   #excess return
    rx_quarterly    = r_quarterly.sub( shy_quarterly, axis=0 )

    rp      = rx_quarterly.mean()  
    vol     = rx_daily.std() * (252/4)**0.5 
    cov     = rx_daily.cov() * (252/4)
    
    w_equal = pd.Series(1/len(assets), index=assets)         #equal weight
    w_equal = np.array(w_equal)
    equal_weight = pd.DataFrame(index=[start])                        
    equal_weight['RP']     = w_equal.dot(rp)
    equal_weight['Vol']    = w_equal.dot(cov).dot(w_equal) ** 0.5      
    equal_weight['Sharpe'] = equal_weight.RP / equal_weight.Vol
   
    return equal_weight.Sharpe

def mcap_weight_sharpe_ratio(start, end, assets):
    r_daily   = PRICE.loc[start:end,assets].ffill().pct_change()
    r_annual  = r_daily[end:end]
    shy_daily = shy[start:end].ffill().pct_change()
    shy_annual= shy_daily[end:end]
    rx_daily  = r_daily .sub( shy_daily, axis=0 )
    rx_annual = r_annual.sub( shy_annual, axis=0 )

    rp      = rx_annual.mean()  
    vol     = rx_daily.std() * (252/4)**0.5 
    cov     = rx_daily.cov() * (252/4)
    
    w_mcap = MCAP.loc[start,assets] / MCAP.loc[start,assets].sum()  #market capital weight
    w_mcap = np.array(w_mcap)
    mcap_weight = pd.DataFrame(index=[start])                        
    mcap_weight['RP']     = w_mcap.dot(rp)
    mcap_weight['Vol']    = w_mcap.dot(cov).dot(w_mcap) ** 0.5      
    mcap_weight['Sharpe'] = mcap_weight.RP / mcap_weight.Vol
   
    return mcap_weight.Sharpe

PART 3: Spread¶

Since my strategy is choosing largest 1000 market cap companies, I used the same spread as the Russell 1000 ETF for the spread of my portfolio. For example, IWB, IWF and IWD. On average, the spread is 0.13% of the stock price.

Therefore, I change the formula in calc_return by add 0.065% at the price when we buy and substract 0.065% at the price when we sell.

def calc_returns(start, end, assets):    
    date       = PRICE.loc[start:end,assets].index
    price      = PRICE.loc[start:end,assets].ffill() 
    price_start= PRICE.loc[start:start,assets].ffill().mul(1.00065)  #buy price:add 0.065%
    price_end  = PRICE.loc[end:end,assets].ffill().mul(0.99935)      #sell price:substract 0.065%

    price_without_start_end = price.loc[date[1]:date[-2], assets]
    price_add_start = price_without_start_end.append(price_start)
    price_add_start_end = price_add_start.append(price_end)
    price_add_start_end.sort_index(inplace=True) 

    price_ret = price_add_start_end.pct_change()
    ret=price_ret[1:].fillna(price_ret).add(1).cumprod() 
    return ret

def get_rebalance_dates(frequency):

def run_backtest(frequency):
    rebalance_dates = get_rebalance_dates(frequency)    
    portfolio_value = pd.Series(1,index=[rebalance_dates[0]])

    for i in range(len(rebalance_dates)-1):
        start_date = rebalance_dates[i]
        end_date   = rebalance_dates[i+1]
        assets_we_can_buy = PRICE.loc[start_date].dropna().index

        weights           = select_weights(start_date, assets)
        cum_ret               = calc_returns(start_date, end_date, assets)
        asset_values          = portfolio_value[-1] * weights  * cum_ret  # dollar values in holding period
        portfolio_value       = portfolio_value.append(asset_values.sum(axis=1)) 
    return portfolio_value
